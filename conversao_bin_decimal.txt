import math
binario = input("Digite um número binário: ")
potencias = []
decimal = 0

for digito in binario:
    if digito != '0' and digito != '1':
       print("Número Inválido")
       break
else:
    for potencia in range(len(binario)):
        potencias.append(math.pow(2,int(potencia)))
    potencias.reverse()
    
    for digito in range(len(binario)):
       decimal += int(binario[digito]) * potencias[digito]
    print(int(decimal))
